import numpy as np
from matplotlib import pyplot as plt 
import cv2
# to make basic image processing functions such as translation, rotation, resizing, 
#skeletonization, displaying Matplotlib images, sorting contours, detecting edges, and much more
import imutils 

cap = cv2.VideoCapture(0)

# memasukkan detector wajah
cascade = cv2.CascadeClassifier('cascades\haarcascade_frontalface_alt.xml')

#mambaca gambar untuk bahan filter
cute_ori = cv2.imread("cute.png" , -1)
heart_ori = cv2.imread("heart.png",-1)

def transparentOverlay(src, overlay , pos = (0,0)  , scale = 1):
    overlay = cv2.resize(overlay , (0,0) ,fx = scale , fy = scale)
    h , w , _ =  overlay.shape ## size of foreground image
    rows , cols , _ = src.shape  ## size of background image
    y , x = pos[0] , pos [1]

    for i in range(h):
        for j in range(w):
            if x + i > rows or y + j >=cols:
                continue
            alpha = float(overlay[i][j][3]/255) ##  read the alpha chanel
            src[x+i][y+j] = alpha * overlay[i][j][:3] + (1-alpha) * src[x+i][y+j]
    return src

while cap.isOpened():
    result , frame = cap.read()
    ret, frame = cap.read()
    if ret == False: break
    frame = imutils.resize(frame, width=720)
    
    if result:
        gray = cv2.cvtColor(frame,cv2.COLOR_BGR2GRAY)
        
        faces =cascade.detectMultiScale(gray ,1.3, 5 , 0 , minSize=(120,120) , maxSize=(350,350))
        
        for (x,y,w,h) in faces:
            if h> 0 and w>0:
                # Mengubah gambar input(heart.png) sesuai ukuran wajah yang terdeteksi
                resized_image = imutils.resize(heart_ori, width=w)
                row_image = resized_image.shape[0]
                col_image = w


                #menentukan ketinggian gambar input (heart.png)
                hight_image = row_image // 6

                dif = 0

                # Menyisipkan gambar input(heart.png) yang telah diubah ukuran kewajah
                if y + hight_image - row_image >= 0:

                    # menentukan tempat untuk meletakkan gambar input (heart.png)
                    n_frame = frame[y + hight_image - row_image : y + hight_image,x : x + col_image]


                else:
                    dif = abs(y + hight_image - row_image) 
                    #meletakkan heart.png kedalam frame
                    n_frame = frame[0 : y + hight_image, x : x + col_image]

                #diubah ukuran dan membalikkan 
                mask = resized_image[:, :, 3]
                mask_inv = cv2.bitwise_not(mask)

                bg_black = cv2.bitwise_and(resized_image, resized_image, mask=mask)
                bg_black = bg_black[dif:, :, 0:3]
                bg_frame = cv2.bitwise_and(n_frame, n_frame, mask=mask_inv[dif:,:])

                result = cv2.add(bg_black, bg_frame)
                if y + hight_image - row_image >= 0:
                    frame[y + hight_image - row_image : y + hight_image, x : x + col_image] = result

                else:
                    frame[0 : y + hight_image, x : x + col_image] = result
                
                # Meletekkan posisi dan ukuran cute.png
                cute_symin = int(y+ 0.75 * h/4)
                cute_symax = int(y + 2.5 * h / 4)
                sh_cute = cute_symax - cute_symin

                face_cute_ori = frame [cute_symin:cute_symax , x:x+w]
                cute = cv2.resize(cute_ori , (w , sh_cute) , interpolation= cv2.INTER_CUBIC)
                transparentOverlay(face_cute_ori  , cute)
                
        cv2.imshow("frame" , frame)
        
        #menyimpan gambar
        cv2.imwrite('hasilFilterCute-SriHariyati.png', frame)
        if cv2.waitKey(1)== ord('q'):
            break
